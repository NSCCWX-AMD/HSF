00040 template<typename T>
00041 void Field<T>::initSend()
00042 {
00043         //- create
00044         if(patchTabPtr_)
00045         {
00046                 //- create
00047                 if(!sendBufferPtr_)
00048                 {
00049                         sendBufferPtr_ = new Table<Word, T*>;
00050                 }
00051 
00052                 //-create
00053                 if(!recvBufferPtr_)
00054                 {
00055                         recvBufferPtr_ = new Table<Word, T*>;
00056                 }
00057 
00058                 Table<Word, Patch*>& patches = *patchTabPtr_;
00059                 Table<Word, T*>& sendBuffer = *sendBufferPtr_;
00060                 Table<Word, T*>& recvBuffer = *recvBufferPtr_;
00061                 label nPatches = patches.size();
00062 
00063                 //- create memory for MPI_Request
00064                 if(!sendRecvRequests_)
00065                 {
00066                         sendRecvRequests_ = new MPI_Request[2*nPatches];
00067                 }
00068 
00069                 Table<Word, Patch*>::iterator it = patches.begin();
00070                 for(label i=0; i<nPatches, it!=patches.end(); ++i, ++it)
00071                 {
00072                         Patch& patchI = *(it->second);
00073                         label patchSize = patchI.getSize();
00074                         Word patchName = it->first;
00075                         //- here memory created
00076                         if(!sendBuffer[patchName])
00077                         {
00078                                 sendBuffer[patchName] = new T[patchSize*ndim_];
00079                         }
00080                         if(!recvBuffer[patchName])
00081                         {
00082                                 recvBuffer[patchName] = new T[patchSize*ndim_];
00083                         }
00084 
00085                         T* patchI_sendBuffer = sendBuffer[patchName];
00086                         T* patchI_recvBuffer = recvBuffer[patchName];
00087 
00088                         label* patchI_addressing = patchI.getAddressing();
00089 
00090                         for(label j=0; j<patchSize; ++j)
00091                         {
00092                                 for(label k=0; k<ndim_; ++k)
00093                                 {
00094                                         patchI_sendBuffer[j*ndim_+k] = localData_[patchI_addressing[j]*ndim_+k];
00095                                 }
00096                         }
00097 
00098                         MPI_Isend
00099                         (
00100                                 patchI_sendBuffer,
00101                                 patchSize*ndim_,
00102                                 MPI_SCALAR,
00103                                 patchI.getNbrID(),
00104                                 1,
00105                                 MPI_COMM_WORLD,
00106                                 &sendRecvRequests_[i]
00107                         );
00108 
00109                         MPI_Irecv
00110                         (
00111                                 patchI_recvBuffer,
00112                                 patchSize*ndim_,
00113                                 MPI_SCALAR,
00114                                 patchI.getNbrID(),
00115                                 1,
00116                                 MPI_COMM_WORLD,
00117                                 &sendRecvRequests_[i + nPatches]
00118                         );
00119                 }
00120         }
00121 }
